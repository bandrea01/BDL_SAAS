version: '3.9'

services:
  restapi:
    container_name: restapi
    image: localhost:5050/restapi
    build: .
    restart: unless-stopped
    ports:

      - DEBUG_PWD=restapi
    depends_on:
      - db


  db:
    container_name: db
    image: postgres:latest
    ports:
      - "5432:5432"
    env_file:
      - .env
    volumes:
      - pgdata:/var/lib/postgresql/data
